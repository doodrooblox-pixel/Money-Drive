        const before=snap(data);
        const tx={id:String(Date.now()),type,amount:type==="expense"?-amt:amt,reason:reason.trim(),date:Date.now()};

    function App() {
      const [screen,setScreen]=useState("loading");
      const [authMode,setAuthMode]=useState("login");
      const [form,setForm]=useState({username:"",password:""});
      const [formErrors,setFormErrors]=useState({});
      const [currentUser,setUser]=useState(null);
      const [userData,setUserData]=useState(null);
      const [toastMsg,setToastMsg]=useState(null);
      const [undoItem,setUndoItem]=useState(null);
      const undoTimer=useRef(null);

      const toast=(type,text)=>{setToastMsg({type,text});setTimeout(()=>setToastMsg(null),4000);};
      const pushUndo=(label,fn)=>{if(undoTimer.current)clearTimeout(undoTimer.current);setUndoItem({label,fn});undoTimer.current=setTimeout(()=>setUndoItem(null),6500);};
      const doUndo=()=>{if(undoItem){undoItem.fn();setUndoItem(null);}if(undoTimer.current)clearTimeout(undoTimer.current);};
      const clearUndo=()=>{setUndoItem(null);if(undoTimer.current)clearTimeout(undoTimer.current);};

      useEffect(()=>{
        const sess=store.get("mm_session_v1");
        if(sess?.userId){const users=store.get("mm_users_v1")||[];const u=users.find(x=>x.id===sess.userId);if(u){setUser(u);const d=store.get(`mm_data_v1_${u.id}`)||emptyData();setUserData(d);setScreen("dashboard");return;}}
        setScreen("auth");
      },[]);

      const handleAuth=()=>{
        const e={};
        if(!form.username.trim())e.username="Username is required";
        if(!form.password||form.password.length<4)e.password="Minimum 4 characters";
        if(Object.keys(e).length){setFormErrors(e);return;}
        const users=store.get("mm_users_v1")||[];
        if(authMode==="signup"){
          if(users.find(u=>u.username.toLowerCase()===form.username.toLowerCase())){setFormErrors({username:"Username already taken"});return;}
          const newUser={id:`${Date.now()}_${Math.random().toString(36).slice(2,8)}`,username:form.username.trim(),password:hashPass(form.password)};
          store.set("mm_users_v1",[...users,newUser]);store.set(`mm_data_v1_${newUser.id}`,emptyData());store.set("mm_session_v1",{userId:newUser.id});
          setUser(newUser);setUserData(emptyData());
        } else {
          const found=users.find(u=>u.username.toLowerCase()===form.username.toLowerCase()&&u.password===hashPass(form.password));
          if(!found){setFormErrors({password:"Incorrect username or password"});return;}
          store.set("mm_session_v1",{userId:found.id});setUser(found);const d=store.get(`mm_data_v1_${found.id}`)||emptyData();setUserData(d);
        }
        setScreen("dashboard");toast("ok",`Welcome, ${form.username.trim()}! 👋`);
      };

      const saveData=(newData)=>{setUserData(newData);if(currentUser)store.set(`mm_data_v1_${currentUser.id}`,newData);};
      const handleLogout=()=>{store.set("mm_session_v1",null);setUser(null);setUserData(null);setForm({username:"",password:""});setFormErrors({});clearUndo();setScreen("auth");};

      if(screen==="loading"){
        setTimeout(()=>setScreen("auth"),600);
        return (<div style={{display:"flex",flexDirection:"column",alignItems:"center",justifyContent:"center",minHeight:"100vh",background:K.bg,gap:16}}><style>{GCSS}</style><div style={{fontFamily:"'Playfair Display'",fontSize:64,color:K.gold,lineHeight:1,animation:"pulse 1.8s ease infinite"}}>₹</div><div style={{fontSize:11,color:K.ts,letterSpacing:"0.22em",textTransform:"uppercase"}}>Loading Money Drive</div></div>);
      }
      if(screen==="auth") return <AuthScreen mode={authMode} form={form} setForm={setForm} errors={formErrors} onSubmit={handleAuth} onToggle={()=>{setAuthMode(m=>m==="login"?"signup":"login");setFormErrors({});}}/>;
      return <Dashboard user={currentUser} data={userData} saveData={saveData} onLogout={handleLogout} toastMsg={toastMsg} toast={toast} undoItem={undoItem} pushUndo={pushUndo} doUndo={doUndo} clearUndo={clearUndo}/>;
    }

    ReactDOM.createRoot(document.getElementById("root")).render(<App/>);
  
</script>
